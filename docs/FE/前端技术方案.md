## 🎯 方案概述

### 1.1 核心要求实现
基于用户明确要求，本方案专注于：
1. **前端源码直接服务**：不编译打包，源码直接提供服务
2. **双端分离架构**：用户端（移动H5）和管理端（PC）分离设计
3. **硬件集成范围**：前端仅负责唤起移动端设备的麦克风和摄像头
4. **前后端同仓库**：前端代码和后端代码放在同一仓库中

### 1.2 技术架构特点（符合编码规范）
- **用户端**：移动H5优先，Vue 2 + Vant 2，硬件唤起功能
- **管理端**：PC优先，Vue 2 + Element UI，运营管理功能
- **无构建架构**：直接通过浏览器运行，无Webpack/Vite构建
- **CDN引入**：所有第三方库通过CDN引入，无本地打包
- **原生JavaScript**：使用原生JS开发，无TypeScript编译
- **组件分离**：组件使用`.html`（模板）+ `.js`（逻辑）双文件结构
- **硬件集成**：前端仅负责唤起，后端处理识别和分析

## 📱 移动端硬件唤起

### 4.1 摄像头唤起实现（符合编码规范）
前端仅负责唤起摄像头，获取视频流：

```javascript
// /utils/hardware.js - 摄像头工具函数
const CameraUtils = {
  cameraStream: null,
  
  /**
   * 唤起摄像头
   * @param {string} mode - 'document'文档模式，'motion'运动模式
   * @returns {Promise<MediaStream>} 视频流
   */
  activateCamera: function(mode = 'document') {
    return new Promise((resolve, reject) => {
      try {
        // 检查浏览器支持
        if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
          reject(new Error('浏览器不支持摄像头访问'));
          return;
        }
        
        // 设置摄像头约束
        const constraints = {
          video: {
            facingMode: 'environment', // 使用后置摄像头
            width: { ideal: mode === 'document' ? 1920 : 1280 },
            height: { ideal: mode === 'document' ? 1080 : 720 },
            frameRate: mode === 'motion' ? { ideal: 30 } : undefined
          }
        };
        
        // 申请摄像头权限
        navigator.mediaDevices.getUserMedia(constraints)
          .then(stream => {
            this.cameraStream = stream;
            resolve(stream);
          })
          .catch(error => {
            reject(this.handleCameraError(error));
          });
      } catch (error) {
        reject(this.handleCameraError(error));
      }
    });
  },
  
  /**
   * 拍照（获取当前帧）
   * @param {HTMLVideoElement} videoElement - 视频元素
   * @returns {string} base64格式的图片数据
   */
  takePhoto: function(videoElement) {
    if (!videoElement || !this.cameraStream) {
      throw new Error('摄像头未激活或视频元素无效');
    }
    
    const canvas = document.createElement('canvas');
    canvas.width = videoElement.videoWidth;
    canvas.height = videoElement.videoHeight;
    
    const ctx = canvas.getContext('2d');
    if (!ctx) {
      throw new Error('无法创建画布上下文');
    }
    
    // 绘制当前视频帧
    ctx.drawImage(videoElement, 0, 0, canvas.width, canvas.height);
    
    // 转换为base64格式（前端仅负责格式转换，识别由后端完成）
    return canvas.toDataURL('image/jpeg', 0.8);
  },
  
  /**
   * 停止摄像头
   */
  deactivateCamera: function() {
    cameraStream.value?.getTracks().forEach(track => track.stop());
    cameraStream.value = null;
  }
  
  return { cameraStream, activateCamera, takePhoto, deactivateCamera };
}
```

### 4.2 麦克风唤起实现
```typescript
// /web/client/composables/hardware/useMicrophone.ts
export function useMicrophone() {
  const microphoneStream = ref<MediaStream | null>(null);
  
  async function activateMicrophone() {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({
        audio: {
          echoCancellation: true,
          noiseSuppression: true,
          sampleRate: 48000
        }
      });
      microphoneStream.value = stream;
      return stream;
    } catch (error) {
      throw handleMicrophoneError(error);
    }
  }
  
  // 录制音频
  function recordAudio(duration: number = 5000): Promise<Blob> {
    return new Promise((resolve, reject) => {
      if (!microphoneStream.value) {
        reject(new Error('麦克风未激活'));
        return;
      }
      
      const mediaRecorder = new MediaRecorder(microphoneStream.value);
      const audioChunks: Blob[] = [];
      
      mediaRecorder.ondataavailable = (event) => {
        audioChunks.push(event.data);
      };
      
      mediaRecorder.onstop = () => {
        resolve(new Blob(audioChunks, { type: 'audio/webm' }));
      };
      
      mediaRecorder.start();
      setTimeout(() => mediaRecorder.stop(), duration);
    });
  }
  
  return { microphoneStream, activateMicrophone, recordAudio };
}
```

// /web/client/utils/device-detector.ts
export class DeviceDetector {
  // 检测摄像头支持
  static async checkCameraSupport() {
    try {
      const devices = await navigator.mediaDevices.enumerateDevices();
      const videoDevices = devices.filter(d => d.kind === 'videoinput');
      return { supported: videoDevices.length > 0, devices: videoDevices };
    } catch {
      return { supported: false, devices: [] };
    }
  }
  
  // 检测浏览器信息
  static getDeviceInfo() {
    const ua = navigator.userAgent;
    return {
      isMobile: /iPhone|iPad|iPod|Android/i.test(ua),
      isIOS: /iPhone|iPad|iPod/i.test(ua),
      isAndroid: /Android/i.test(ua),
      isWeChat: /MicroMessenger/i.test(ua)
    };
  }
}

### 6.3 错误处理和状态管理
```typescript
// 统一的API错误处理
import axios from 'axios';

const api = axios.create({
  baseURL: '/api',
  timeout: 10000
});

// 请求拦截器
api.interceptors.request.use(config => {
  // 添加认证token
  const token = localStorage.getItem('token');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  
  // 添加租户ID
  const tenantId = localStorage.getItem('tenant_id');
  if (tenantId) {
    config.headers['X-Tenant-ID'] = tenantId;
  }
  
  return config;
});

// 响应拦截器
api.interceptors.response.use(
  response => response.data,
  error => {
    // 统一错误处理
    if (error.response?.status === 401) {
      // 未授权，跳转到登录页
      window.location.href = '/login';
    } else if (error.response?.status === 403) {
      // 权限不足
      showErrorMessage('权限不足，请联系管理员');
    } else if (error.response?.status === 429) {
      // 请求过于频繁
      showErrorMessage('操作过于频繁，请稍后重试');
    } else {
      // 其他错误
      showErrorMessage(error.response?.data?.message || '网络错误');
    }
    
    return Promise.reject(error);
  }
);
```

### 7.1 方案优势
1. **源码直接服务**：无需构建打包，开发部署简单
2. **双端分离设计**：用户端和管理端针对不同场景优化

5. **CDN加速**：核心库使用CDN，提高加载速度

4. **硬件API标准化**：使用WebRTC标准API，兼容性好
